# Create the menu
wp menu create "JPNDC Top Menu"

wp menu item list jpndc-top-menu

# Add Donate (External, New Tab)
ID_DONATE=$(wp menu item add-custom jpndc-top-menu "Donate" "https://giving.classy.org/campaign/748362/donate" --porcelain)
wp post meta update $ID_DONATE _menu_item_target "_blank"
wp post meta update $ID_DONATE _menu_item_xfn "nofollow noopener"

# Add Contact Us (Internal Page ID 25)
wp menu item add-post jpndc-top-menu 25 --title="Contact Us"

# Add Language Dropdown Placeholder
# This creates the "English / Español" trigger
ID_LANG=$(wp menu item add-custom jpndc-top-menu "English" "#" --porcelain)
wp post meta update $ID_LANG _menu_item_classes '["lang-item"]'

# 5. Assign to Avada Location
wp menu location assign jpndc-top-menu top_navigation
